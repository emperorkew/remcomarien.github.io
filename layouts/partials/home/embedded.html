void sensor_read_loop(void) {
  float temp = pt100_read(ADC_CH1);
  float pres = transducer_read(ADC_CH2);

  // Apply 1st-order digital filter
  temp = alpha * temp + (1.0f - alpha) * prev_t;
  prev_t = temp;

  // Write to MODBUS holding registers
  mb_regs[0] = float_to_u16(temp);
  mb_regs[1] = float_to_u16(pres);

  HAL_GPIO_TogglePin(LED_PORT, LED_PIN);
}
